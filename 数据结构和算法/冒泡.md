const arr= [1, 8, -1, 120, 10, 88]

function maopao(arr) {
  const len = arr.length - 1;
  for (let i = 0; i < len; i++) {

    for(let j = 0; j < len - i; j++) {
      let temp;
      if (arr[j] > arr[j + 1]) {
        temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp
      }

    }

  }
  return arr
}
